Diese ist wie folgt:
\begin{itemize}
    \item Betreuer der Arbeit
    \item Weiteres Lehrpersonal
    \item Technisches Personal
    \item Institutionen
    \item Interviewpartner
    \item Lektor
    \item Freunde und Familie
    \item Widmung
\end{itemize}
